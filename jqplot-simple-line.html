A basic, responsive line chart using the jqPlot library, with hardcoded data values in five categories, rather than specific data points. Good for when the x axis isn't a discrete value (2.5) and is something like a whole month or year.

require(
	['jquery','jquery-ui','lodash','vendor/plugins/jqplot'],

	function($, _) {
		var ticks = [2010,2011,2012,2013,2014];
		var yvalues = [2,15,50,45,3];
		var options = {
				grid: {
					drawBorder: false, 
					drawGridlines: false,
					background: '#ffffff',
					shadow:false
				},
				axesDefaults: {
					labelRenderer: $.jqplot.CanvasAxisLabelRenderer
				},
				seriesDefaults: {
					color: '#2594E3',
					shadow:false,
					pointLabels: {
						show: true,
						formatString: "%'.0f",//this will display labels with commas, no decimals
						ypadding: 10
					},
					markerOptions: {
						shadow:false
					},
					breakOnNull: true
				},
				axes: {
					xaxis: {
						renderer: $.jqplot.CategoryAxisRenderer,
						ticks: ticks,
						label: "Year",
						drawBaseline:true,
					},
					yaxis: {
						label: "Some quantity of stuff",
						pad:1.2,
						min:0,
						tickOptions:{
							formatString: "%'.0f"
						}
					}
				}
			};
			var plot1 = $.jqplot('line-chart1',[yvalues],options);
			$(window).resize('resize',function() {
	    		plot1.replot({resetAxes:['xaxis']});
			});

	}//end main function
);//end require
