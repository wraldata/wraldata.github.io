require(
	['jquery','lodash','vendor/plugins/jqplot'],

	function($, _) {

		var data = [24967702.07,13308801.05,269261.31,45429.29];
		var ticks = ['Thing 1','Thing 2','Thing 3','Thing 4'];

		var options = { 
			seriesColors: ['#a6cee3', '#b2df8a','#33a02c','#1f78b4'],
			grid: {
				drawBorder: false, 
				drawGridlines: false,
				background: '#ffffff',
				shadow:false
			},
			seriesDefaults: {
				renderer: $.jqplot.BarRenderer,
				rendererOptions:{
					varyBarColor: true,
				},
				shadow:false,
				pointLabels: {
					show: true,
					formatString: "$%'i"
				}
			},
			axes: {
				xaxis: {
					renderer: $.jqplot.CategoryAxisRenderer,
					ticks: ticks
				},
				yaxis: {
					tickOptions: {
						formatString: "$%'i"
					},
					pad: 1.2,
					min: 0
				}
			},
			highlighter: {
				show: false
			}
		};
		var plot1 = $.jqplot ('bar-chart1', [data], options);
		$(window).resize('resize',function() {
    		plot1.replot({resetAxes:['xaxis']});
		});
	}
);
